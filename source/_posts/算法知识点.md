## 1.algorithm库的使用

```c++
#include<iostream>
#include<algorithm>
#include<cstring>
#include<sstream>

using namespace std;

typedef long long LL;

const int N=1e5+10;


int main()
{
    // 1.max()、min()、abs()函数
    
    int a = 3, b = 4;
    //求最大值
    int Max = max(a,b);
    //求最小值
    int Min = min(a,b);
    //求绝对值
    int Abs = abs(-3);
    cout << Max << Min << Abs;
    
    // 2.交换函数：swap()
    
    int a = 3, b = 4;
    swap(a,b);
    cout << a << b;
    return 0;
    
    // 3.翻转函数：reverse()
    
    int a[5] = {11,22,33,44,55};
    reverse(a,a+5);
    for(int i = 0; i < 5; i++) 
        cout << a[i] << ' ';
    return 0;

    // 4.排序函数：sort()
    
    int a[5] = {55,44,33,22,11};
	
    sort(a,a+5);
	
    for(int i = 0; i < 5; i++) 
        cout << a[i] << ' ';
        
    // 5.填充函数：fill()
    
    int a[5] = {11,33,22};
	
    fill(a+3,a+5,9999);								
	
    for(int i = 0; i < 5; i++) 
        cout << a[i] << ' ';
    return 0;

    // 6.C++新语法：
    // count(a, b, c) (a, b)的区间内c的个数
    // find(a, b, c) 在(a, b)区间内找c并返回其指针

    cin>>n;
    for(int i=1;i<=n;i++) cin>>a[i];
	
    if(count(a+1,a+n+1,a[1])==n){
        cout<<0<<'\n';
        return;
    }
	
    if(count(a+1,a+n+1,1)>=1){
        cout<<-1<<'\n';
        return;
    }
	
    if(count(a+1,a+n+1,a[1])<n){
        int t=find(a+1,a+n+1,2)-a;
        for(int i=1;i<=n;i++){
            while(a[i]!=2){
                a[i]=(a[i]+1)/2;
                c.push_back({i,t});
            }
        }
    }
	
    return 0;
}
```

## 2. cstring 库使用
```c++
#include<iostream>
#include<algorithm>
#include<cstring>                  //头文件

using namespace std;

const int N=1e5+10;

int h[N],g[N];
int s[N][N];
int n;


int main()
{
    cin>>n;
    for(int i=0;i<n;i++) cin>>g[i];
    
    memset(g,1,sizeof g)           //将g数组每一位都填充1（用cstring库）
    memset(s,1,sizeof s)           //二维填充
    
    fill(g,g+N,1)                  //同上，填充数组（用的algorithm库）
    fill(s[0],s[0]+N*N,1)          //二维填充
    
    memcpy(h,g,sizeof g)           //复制g数组中每一位到h数组中（复制备份数组）
    
    return 0;
}
```

## 5.sstream库的使用
```c++
#include<iostream>
#include<algorithm>
#include<cstring>
#include<sstream>

using namespace std;

typedef long long LL;

const int N=1e5+10;

int n;
int a[N];


int main()
{
    int cnt;
    cin>>cnt;
    
    string line;
    getline(cin,line);               //读第一行回车（没有可以不用写）
	
    while(cnt--){
        getline(cin,line);           //过滤每一行的换行符
        
        stringstream ssin(line);     //用ssin代替cin读入每一行
        
        while(ssin>>a[n]) n++;       //把数字提取出来存进数组
    }
	
    return 0;
}
```
